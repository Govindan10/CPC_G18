 vector<int> twoRepeated(vector<int>& arr) {
  
        int n=arr.size();
        int x=-1,y=-1;
        for(int i=0;i<n;i++){
            if(arr[abs(arr[i])]>0){
                arr[abs(arr[i])]=-arr[abs(arr[i])];
            }
            else{
                if(x==-1)x=abs(arr[i]);
                else y=abs(arr[i]);
            }
            
        }
        return {x,y};
    }
