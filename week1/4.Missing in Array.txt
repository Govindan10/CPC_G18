 int missingNum(vector<int>& arr) {
        // code here
        int n=arr.size();
        int xor_sum=0;
        for(int i=0;i<n;i++){
            xor_sum^=arr[i];
        }

        if((n+1)%4==0)return xor_sum^(n+1);
        else if((n+1)%4==1)return xor_sum^(1);
        else if((n+1)%4==2)return xor_sum^(n+2);
        return xor_sum;
    }
