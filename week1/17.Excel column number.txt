int titleToNumber(string columnTitle) {
        long ans=0;
        int n=columnTitle.size();
        ans+=(26*(pow(26,n-1)-1)/25);
        
        int x=n-1;
        for(int i=0;i<n;i++){
            ans+=(pow(26,x)*(columnTitle[i]-'A'));
            x--;
        }
        return ans+1;
    }
