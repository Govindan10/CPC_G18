int findDuplicate(vector<int>& arr) {
        int n=arr.size();
        for(int i=0;i<n;i++){
            int pos=abs(arr[i])-1;
            if(arr[pos]>0){
                arr[pos]=-arr[pos];
            }
            else{
                return abs(arr[i]);
            }
        }
        return -1;
    }
