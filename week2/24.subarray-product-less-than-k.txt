int numSubarrayProductLessThanK(vector<int>& nums, int k) {
        int prod=1;
        int n=nums.size();
        int j=0;
        int count=0;
        for(int i=0;i<n;i++){
            prod*=nums[i];
            while(prod>=k && j<=i){
                prod=prod/nums[j++];
            }
            if(j<=i)count+=(i-j+1);
        }
        return count;
    }
