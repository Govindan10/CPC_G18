bool twoSum(vector<int>& arr, int target) {
        // code here
        unordered_map<int,int> mpp;
        mpp[0]++;
        for(int i=0;i<arr.size();i++){
            int rem=target-arr[i];
            if(rem==0){if(mpp[0]>1)return 1;}
            else if(mpp[rem]){
                return true;
            }
            mpp[arr[i]]++;
        }
        return 0;
    }
