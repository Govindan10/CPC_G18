int maxAscendingSum(vector<int>& nums) {
        int n=nums.size();
      
        int prev=-1;
        int sum=0,maxsum=0;
        for(int r=0;r<n;r++){
            if(nums[r]<=prev){
                sum=0;
           
            }
            sum+=nums[r];
            maxsum=max(maxsum,sum);
            prev=nums[r];
        }
        return maxsum;
    }
