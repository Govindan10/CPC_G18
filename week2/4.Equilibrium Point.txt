int findEquilibrium(vector<int> &arr) {
     
        int n=arr.size();
        vector<int> prefixSum(n,0);
        prefixSum[0]=arr[0];
        int total=0;
        for(int i=1;i<n;i++){
            total=prefixSum[i]=prefixSum[i-1]+arr[i];
        }
        for(int i=0;i<n;i++){
            int s1=prefixSum[i]-arr[i];
            int s2=total-prefixSum[i];
            if(s1==s2)return i;
        }
        return -1;
    }
